void TurtleMobility::resumeScript ( ) [protected, virtual]

Process next statements from script.

Will set a new nextChange and targetPosition.

References executeStatement(), gotoNextStatement(), MobilityBase::lastPosition, MovingMobilityBase::nextChange, MovingMobilityBase::stationary, and LineSegmentsMobilityBase::targetPosition.

Referenced by initializePosition(), and setTargetPosition().

{
    simtime_t now = simTime();

    do {
        if (nextStatement != NULL)
        {
            executeStatement(nextStatement);
            gotoNextStatement();
        }
        else
        {
            nextChange = -1;
            stationary = true;
            targetPosition = lastPosition;
        }
    } while (nextChange == now);
}
