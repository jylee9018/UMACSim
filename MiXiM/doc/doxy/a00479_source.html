00057   virtual simtime_t switchTo(int newState, simtime_t_cref now) {
00058     // state must be one of sleep, receive or transmit (not sync)
00059     //assert(newState != Radio::SYNC);
00060     if(newState == state || (newState == RadioUWBIR::RX && state == RadioUWBIR::SYNC)) {
00061       return -1; // nothing to do
00062     } else {
00063       if(newState == RadioUWBIR::RX) {
00064         // prevent entering "frame reception" immediately
00065         newState = RadioUWBIR::SYNC;
00066       }
00067       return reallySwitchTo(newState, now);
00068     }
00069   }
00070 
00071   virtual simtime_t reallySwitchTo(int newState, simtime_t_cref now) {
00072     // set the nextState to the newState and the current state to SWITCHING
00073     nextState = newState;
00074     int lastState = state;
00075     state = RadioUWBIR::SWITCHING;
00076     radioStates.record(state);
00077     // make entry to RSAM
00078     makeRSAMEntry(now, state);
00079 
00080     // return matching entry from the switch times matrix
00081     return swTimes[lastState][nextState];
00082   }
